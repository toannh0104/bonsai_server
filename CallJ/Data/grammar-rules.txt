# Grammar Rule Definition file

# ==========================================================================
#
# Notes on Format
#
# <key> 	This is a command that will be interpreted by the parser
# [key] 	Takes whatever is currently set in the resolved slot-filler for the component "key"
# {key} 	Refers to the grammar rule referred to by "key". Also used for counters
# (key) 	Refers to a link to the lexicon, and would include word type, the word label, and any 
#         	other information on form, etc. 
# 
# |key|	A variable
#
#
# A note on the component costs
# ==========================
# These are set by a number after each component in a template.
# 5 is high cost, 1 is low. 0 means no cost associated with this component.
# -1 is a wildcard, accepting a cost pressumaby set at a higher lever (functions)
# If not set, defaults to 3.
#

<rule DESU_SIMPLE1>
	<form verbForm>
		<setting>
			present pos neg polite statement		
			<condition [Verb] == is_also>
				FORM neg off
			</condition>
		</setting>
		<setting>
			present pos polite question		
			<condition [Subject] =group= ingroup>
				FORM question off
				FORM statement on
				FORM neg on
			</condition>
		</setting>
	</form>
	<template>
		{Subject 1 4} [Complement 1 4] {Verb 4 2} 
	</template>
	<rule Subject>
		<template>
			[Subject -1 -1] {Particle 3 1}
			<restriction [Subject] =group= person>
				{Person -1 -1 [Subject]} {Particle 3 1}
			</restriction>
			<restriction [Subject] =group= definitiveA>
				[Subject 4 2] {Particle 3 1}
			</restriction>
		</template>
		<rule Particle>
			<template>
				{ParticleSUBJ -1 -1}
				<restriction [Verb] == is_also>
					(particle も -1 -1)
				</restriction>				
			</template>
		</rule>
	</rule>
	<rule Verb>
		<template>
			<vrule basic verbForm>			
				(verb です -1 -1) 
			</vrule>
		</template>	
	</rule>
</rule>
